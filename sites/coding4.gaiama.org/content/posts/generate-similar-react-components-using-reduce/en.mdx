```js
const elements = [
  [`h1`, ``],
  [`h2`, ``],
  [`h3`, ``],
  [`h4`, ``],
  [`h5`, ``],
  [`h6`, ``],
].reduce(
  (acc, [Tag, classes]) => ({
    ...acc,
    [Tag]: props => (
      <Tag {...(classes ? { className: classes } : {})} {...props} />
    ),
  }),
  {}
)
```
